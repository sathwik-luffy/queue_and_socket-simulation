Server Python Code
import socket server = socket.socket() server.bind(('0.0.0.0', 8080)) server.listen(5) conn, addr = server.accept() while True: data = conn.recv(1024) if not data: break conn.send(b"ACK: " + data)
Client Python Code
import socket client = socket.socket() client.connect(('127.0.0.1', 8080)) msg = "Hello Server" client.send(msg.encode()) response = client.recv(1024) print(response.decode()) # → "ACK: Hello Server"
